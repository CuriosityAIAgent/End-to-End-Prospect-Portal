  <script>
    // ============================================================================
    // STORAGE
    // ============================================================================

    const STORAGE_KEYS = {
      draft: 'pb-filenote-draft',
      scanResult: 'pb-filenote-scanresult',
      apiKey: 'pb-filenote-apikey'
    };

    function getApiKey() {
      return localStorage.getItem(STORAGE_KEYS.apiKey);
    }

    function saveApiKey(key) {
      localStorage.setItem(STORAGE_KEYS.apiKey, key);
    }

    function getDraft() {
      const draft = localStorage.getItem(STORAGE_KEYS.draft);
      return draft ? JSON.parse(draft) : {
        contact: '',
        meetingType: 'Prospect first meeting',
        date: new Date().toISOString().split('T')[0],
        noteText: '',
        lastSaved: null
      };
    }

    function saveDraft(contact, meetingType, date, noteText) {
      const draft = {
        contact,
        meetingType,
        date,
        noteText,
        lastSaved: new Date().toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })
      };
      localStorage.setItem(STORAGE_KEYS.draft, JSON.stringify(draft));
      return draft.lastSaved;
    }

    function getScanResult() {
      const result = localStorage.getItem(STORAGE_KEYS.scanResult);
      return result ? JSON.parse(result) : null;
    }

    function saveScanResult(result) {
      localStorage.setItem(STORAGE_KEYS.scanResult, JSON.stringify(result));
    }

    // ============================================================================
    // DEBOUNCE
    // ============================================================================

    function debounce(func, delay) {
      let timeoutId;
      return function(...args) {
        clearTimeout(timeoutId);
        timeoutId = setTimeout(() => func(...args), delay);
      };
    }

    // ============================================================================
    // UI INITIALIZATION
    // ============================================================================

    const setupScreen = document.getElementById('setupScreen');
    const apiKeyInput = document.getElementById('apiKeyInput');
    const saveApiKeyBtn = document.getElementById('saveApiKeyBtn');
    const contactInput = document.getElementById('contactInput');
    const meetingTypeSelect = document.getElementById('meetingTypeSelect');
    const meetingDateInput = document.getElementById('meetingDateInput');
    const noteTextarea = document.getElementById('noteTextarea');
    const saveStatus = document.getElementById('saveStatus');
    const wordCount = document.getElementById('wordCount');
    const scanBtn = document.getElementById('scanBtn');
    const checkPanelEmpty = document.getElementById('checkPanelEmpty');
    const resultsContainer = document.getElementById('resultsContainer');
    const copyNoteBtn = document.getElementById('copyNoteBtn');
    const printNoteBtn = document.getElementById('printNoteBtn');
    const previewPanelEmpty = document.getElementById('previewPanelEmpty');
    const rewritePreview = document.getElementById('rewritePreview');
    const previewContent = document.getElementById('previewContent');
    const acceptRewriteBtn = document.getElementById('acceptRewriteBtn');
    const retryRewriteBtn = document.getElementById('retryRewriteBtn');
    const detailsPanel = document.getElementById('detailsPanel');
    const enhanceBtn = document.getElementById('enhanceBtn');

    // Initialize on page load
    function initializeApp() {
      if (!getApiKey()) {
        setupScreen.classList.add('active');
      }
      loadDraft();
      loadScanResult();
      updateWordCount();
    }

    saveApiKeyBtn.addEventListener('click', () => {
      const key = apiKeyInput.value.trim();
      if (!key) return;
      saveApiKey(key);
      setupScreen.classList.remove('active');
    });

    apiKeyInput.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') saveApiKeyBtn.click();
    });

    // Load draft data
    function loadDraft() {
      const draft = getDraft();
      contactInput.value = draft.contact;
      meetingTypeSelect.value = draft.meetingType;
      meetingDateInput.value = draft.date;
      noteTextarea.value = draft.noteText;
      if (draft.lastSaved) {
        saveStatus.textContent = `Last saved: ${draft.lastSaved}`;
        saveStatus.classList.remove('unsaved');
      }
    }

    // Load scan result if exists
    function loadScanResult() {
      const result = getScanResult();
      if (result) {
        previewContent.textContent = result;
        previewPanelEmpty.style.display = 'none';
        rewritePreview.style.display = 'flex';
      }
    }

    // ============================================================================
    // AUTO-SAVE
    // ============================================================================

    const debouncedSave = debounce(() => {
      const lastSaved = saveDraft(
        contactInput.value,
        meetingTypeSelect.value,
        meetingDateInput.value,
        noteTextarea.value
      );
      saveStatus.textContent = `Last saved: ${lastSaved}`;
      saveStatus.classList.remove('unsaved');
    }, 600);

    function markAsUnsaved() {
      saveStatus.textContent = 'Unsaved changes';
      saveStatus.classList.add('unsaved');
      debouncedSave();
    }

    contactInput.addEventListener('input', markAsUnsaved);
    meetingTypeSelect.addEventListener('change', markAsUnsaved);
    meetingDateInput.addEventListener('change', markAsUnsaved);
    noteTextarea.addEventListener('input', () => {
      markAsUnsaved();
      updateWordCount();
    });

    // ============================================================================
    // WORD COUNT
    // ============================================================================

    function updateWordCount() {
      const text = noteTextarea.value.trim();

    async function callAntropicAPI(contact, meetingType, date, noteText) {
      const apiKey = getApiKey();

      const systemPrompt = `You are an expert private banking file note writer. Your job is to rewrite raw file notes into polished, professional private banking documentation that follows industry standards.

Your rewritten notes should:
- Use clear, concise business language appropriate for private banking
- Organize information logically and hierarchically
- Follow this structure:
  * Client / Contact: [name and key context]
  * Meeting Type & Date: [type and date]
  * Key Discussion Points: [main topics discussed, organized by theme]
  * Client Profile & Situation: [relevant background, financial position, family situation]
  * Client Objectives & Priorities: [investment goals, risk tolerance, key concerns]
  * Current Relationships: [existing advisers, banking relationships, any competitive threats]
  * Next Steps & Agreed Actions: [specific commitments, follow-ups, responsible parties, timelines]
  * Summary / Key Themes: [brief assessment of relationship trajectory, opportunities, risk factors]
- Maintain all factual information from the original note
- Enhance clarity without adding new information
- Use professional tone suitable for regulatory review
- Be concise but complete
- Format with clear headings and bullet points where appropriate

Return only the reformatted file note. Do not include any preamble, explanation, or markdown formatting outside the note itself.`;

      const userPrompt = `Please rewrite this raw file note into a professional private banking file note format:

Meeting Type: ${meetingType}
Contact: ${contact}
Date: ${date}

Raw Note:
${noteText}`;

      try {
        const requestBody = {
          model: 'claude-opus-4-5',
          max_tokens: 3000,
          system: systemPrompt,
          messages: [
            {
              role: 'user',
              content: userPrompt
            }
          ]
        };

        console.log('Request body:', requestBody);

        const response = await fetch('https://api.anthropic.com/v1/messages', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'x-api-key': apiKey,
            'anthropic-version': '2023-06-01',
            'anthropic-dangerous-direct-browser-access': 'true'
          },
          body: JSON.stringify(requestBody)
        });

        console.log('Response status:', response.status);
        console.log('Response ok:', response.ok);

        if (!response.ok) {
          let errorMsg = `API Error: ${response.status}`;
          try {
            const errorData = await response.json();
            console.error('API Error Response:', errorData);
            errorMsg = errorData.error?.message || errorMsg;
          } catch (e) {
            console.error('Could not parse error response');
          }
          throw new Error(errorMsg);
        }

        const data = await response.json();
        console.log('API Response:', data);
        const content = data.content[0].text;

        return content.trim();
      } catch (error) {
        console.error('Full error details:', {
          name: error.name,
          message: error.message,
          stack: error.stack
        });
        throw error;
      }
    }

    document.getElementById('changeApiKeyBtn').addEventListener('click', () => {

    scanBtn.addEventListener('click', async () => {
      if (!getApiKey()) {
        apiKeyInput.value = '';
        setupScreen.classList.add('active');
        return;
      }

      const noteText = noteTextarea.value.trim();
      const wordCount = noteText.length > 0 ? noteText.split(/\s+/).length : 0;

      if (wordCount < 50) {
        previewContent.textContent = 'Your note is quite short. Add more detail for a professional rewrite.';
        previewPanelEmpty.style.display = 'none';
        rewritePreview.style.display = 'flex';
        acceptRewriteBtn.style.display = 'none';
        retryRewriteBtn.style.display = 'none';
        return;
      }

      scanBtn.disabled = true;
      previewPanelEmpty.style.display = 'none';
      rewritePreview.style.display = 'flex';
      previewContent.textContent = 'Rewriting your note...';
      acceptRewriteBtn.style.display = 'block';
      retryRewriteBtn.style.display = 'block';

      try {
        const result = await callAntropicAPI(
          contactInput.value,
          meetingTypeSelect.value,
          meetingDateInput.value,
          noteText
        );

        if (result) {
          saveScanResult(result);
          previewContent.textContent = result;
          detailsPanel.classList.add('active');
        }
      } catch (error) {
        if (error.message.toLowerCase().includes('api_key') || error.message.toLowerCase().includes('x-api-key') || error.message.toLowerCase().includes('invalid')) {
          localStorage.removeItem(STORAGE_KEYS.apiKey);
          apiKeyInput.value = '';
          setupScreen.classList.add('active');
        }
        previewContent.textContent = `Error: ${error.message}\n\nPlease try again.`;
      } finally {
        scanBtn.disabled = false;
      }
    });
